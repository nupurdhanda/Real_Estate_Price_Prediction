function toggleFields() { var location = document.getElementById("location").value; if (location === "Bengaluru") { document.getElementById("bengaluruFields").style.display = "block"; document.getElementById("mumbaiFields").style.display = "none"; document.getElementById("delhiNCRFields").style.display = "none"; } else if (location === "Mumbai") { document.getElementById("bengaluruFields").style.display = "none"; document.getElementById("mumbaiFields").style.display = "block"; document.getElementById("delhiNCRFields").style.display = "none"; } else if (location === "Delhi NCR") { document.getElementById("bengaluruFields").style.display = "none"; document.getElementById("mumbaiFields").style.display = "none"; document.getElementById("delhiNCRFields").style.display = "block"; } }
